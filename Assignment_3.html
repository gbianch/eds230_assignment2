library(tidyverse)
library(here)

#read in the data 
climate <- read.table(here("data", "clim.txt")) %>% 
  row_to_names(row_number = 1) %>% 
  mutate_all(as.numeric) 
#source function
source(here("R", "almond_modelR.R"))
#run function on climate data 
yield_anomaly <- almond_modelR(clim_data = climate)
#sensitivity analysis for p coefficient

deviation = 0.02
base_pcoeff = 0.07
pcoeff1 <- runif(min=base_pcoeff-deviation*base_pcoeff, 
                max = base_pcoeff+deviation*base_pcoeff, n=20)


pcoeff_sensitivity <- pcoeff1 %>% 
  map(~almond_modelR(clim_data = climate, tmincoeff1 = 0.015, tmincoeff2 = 0.0046, pcoeff1 = .x, pcoeff2 = 0.0043)) 

# extract data
pcoeff_df = map_df(pcoeff_sensitivity,`[`, c("yield", "year", "profit")) %>% 
  mutate(year = as.factor(year))
pcoeff_df$pcoeff = pcoeff1

## sensitivity analysis for price of almonds

price_deviation = 0.05
base_price = 5.95
price_var <- runif(min=base_price-price_deviation*base_price, 
                max = base_price+price_deviation*base_price, n=20)

price_sensitivity <- price_var %>% 
  map(~almond_modelR(clim_data = climate, tmincoeff1 = 0.015, tmincoeff2 = 0.0046, pcoeff1 = 0.07, pcoeff2 = 0.0043, price = .x)) 

price_df <- map_df(price_sensitivity,`[`, c("yield", "year", "profit")) %>% 
  mutate(year = as.factor(year))

#visualize first temperature coefficient sensitivity
ggplot(tcoeff_df, aes(y = profit, x = tcoeff)) +
  geom_point(aes(color = year)) +
  labs(x="temperature coefficient 1")

The figure shows almond profits plotted against the distribution of potential values for the first temperature coefficient in the almond yield model. The figure shows some grouping in profits by year with high profits where expected, such as in 1995 which coincides with high precipitation. However, the high profit years also show a drop off in profit at either the high or low end of the coefficient values, which may suggest that there is another variable we should look at for model sensitivity.

# visualize price sensitivity
ggplot(price_df, aes(y = profit, x = year)) +
  geom_boxplot() +
  labs(x = "year")

The figure shows the profit variability for each year based on changes in price/kg of almonds. The figure suggests that almond price may not be the most sensitive variable in the model and another variable such as precipitation may have a larger impact on the model.
